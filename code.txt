import streamlit as st
from openai import OpenAI
import requests
import re
import pandas as pd
import plotly.express as px
import plotly.graph_objects as go
from io import BytesIO
import yfinance as yf

st.set_page_config(page_title="Financial decision making with ESG", layout="wide")

# 🔑 KEYS
client = OpenAI(api_key="KEY")
NEWS_API_KEY = "KEY"

# ------------------ NEWS ------------------ #
@st.cache_data
def get_news(company):
    try:
        url = f"https://newsapi.org/v2/everything?q={company}&apiKey={NEWS_API_KEY}"
        res = requests.get(url).json()
        return [a["title"] for a in res.get("articles", [])[:5]]
    except:
        return []

# ------------------ SENTIMENT ------------------ #
@st.cache_data
def get_sentiment(news):
    if not news:
        return "No news available", 0

    prompt = f"Analyze sentiment: {news}"

    try:
        res = client.chat.completions.create(
            model="gpt-4.1-mini",
            messages=[{"role":"user","content":prompt}],
            temperature=0
        )
        text = res.choices[0].message.content
        score_match = re.search(r'-?\d+\.?\d*', text)
        score = float(score_match.group()) if score_match else 0
        return text, score
    except:
        return "Sentiment unavailable", 0

# ------------------ ESG ------------------ #
@st.cache_data
def analyze_esg(company, news, sentiment_score):
    prompt = f"ESG analysis for {company} | News: {news} | Sentiment: {sentiment_score} | ESG Score: X/10"

    try:
        res = client.chat.completions.create(
            model="gpt-4.1-mini",
            messages=[{"role":"user","content":prompt}],
            temperature=0
        )
        return res.choices[0].message.content
    except:
        return "ESG Score: 5/10"

# ------------------ HELPERS ------------------ #
def extract_score(text):
    m = re.search(r'(\d+)/10', text)
    return int(m.group(1)) if m else 5

def calculate_risk(score):
    if score <= 4: return "High 🔴"
    elif score <= 7: return "Moderate 🟡"
    else: return "Low 🟢"

def investment_call(score, sentiment):
    if score >= 8 and sentiment > 0.2:
        return "Strong Buy 🚀"
    elif score >= 7:
        return "Buy ✅"
    elif score >= 5:
        return "Hold 🟡"
    elif sentiment < 0:
        return "Reduce ⚠"
    else:
        return "Avoid ❌"

# ------------------ VISUALS ------------------ #
def show_gauge(score):
    return go.Figure(go.Indicator(mode="gauge+number", value=score, gauge={'axis': {'range': [0,10]}}))

def generate_trend(score):
    years = ["2019","2020","2021","2022","2023","2024"]
    values = [max(1, score-3+i) for i in range(6)]
    return px.line(pd.DataFrame({"Year":years,"Score":values}), x="Year", y="Score")

def compare_trend(c1,c2,s1,s2):
    df = pd.DataFrame({
        "Year":["2019","2020","2021","2022","2023","2024"]*2,
        "Score":[s1-3,s1-2,s1-1,s1,s1,s1]+[s2-3,s2-2,s2-1,s2,s2,s2],
        "Company":[c1]*6+[c2]*6
    })
    return px.line(df,x="Year",y="Score",color="Company")

# ------------------ STOCK FIX ------------------ #
def get_ticker(company):
    try:
        search = yf.Tickers(company)
        keys = list(search.tickers.keys())
        if keys:
            return keys[0]
        return company.upper().replace(" ", "") + ".NS"
    except:
        return company

def get_stock_data(company):
    try:
        ticker = get_ticker(company)
        df = yf.Ticker(ticker).history(period="1y")
        return df if not df.empty else None
    except:
        return None

# ------------------ DRIVERS ------------------ #
@st.cache_data
def generate_drivers(company,text):
    res = client.chat.completions.create(
        model="gpt-4.1-mini",
        messages=[{"role":"user","content":f"Top ESG drivers for {company}:\n{text}"}],
        temperature=0
    )
    return res.choices[0].message.content

# ================= UI ================= #

st.title("🌱 ESG Metrics")

company = st.text_input("Enter Company Name")

if st.button("Analyze ESG"):
    if company:
        news = get_news(company)
        sentiment_text, sentiment_score = get_sentiment(news)
        result = analyze_esg(company, news, sentiment_score)
        score = extract_score(result)

        k1,k2,k3 = st.columns(3)
        k1.metric("ESG Score", score)
        k2.metric("Risk", calculate_risk(score))
        k3.metric("Sentiment", round(sentiment_score,2))

        st.plotly_chart(show_gauge(score))
        st.plotly_chart(generate_trend(score))

        st.info(sentiment_text)
        st.write(result)

# ================= COMPARISON ================= #

st.markdown("---")
st.header("🔍 ESG Comparison")

c1 = st.text_input("Company 1")
c2 = st.text_input("Company 2")

if st.button("Compare ESG"):

    if c1 and c2:

        r1 = analyze_esg(c1, get_news(c1), 0)
        r2 = analyze_esg(c2, get_news(c2), 0)

        s1 = extract_score(r1)
        s2 = extract_score(r2)

        # ✅ KPI CARDS
        k1,k2,k3 = st.columns(3)
        k1.metric("Score Difference", abs(s1 - s2))
        leader = c1 if s1 > s2 else c2 if s2 > s1 else "Equal"
        k2.metric("Leader", leader)
        k3.metric("Average ESG", round((s1+s2)/2,2))

        st.divider()

        st.plotly_chart(compare_trend(c1,c2,s1,s2), use_container_width=True)

        # STOCK
        df1 = get_stock_data(c1)
        df2 = get_stock_data(c2)

        if df1 is not None and df2 is not None:

            df1 = df1.reset_index()
            df1["Company"] = c1

            df2 = df2.reset_index()
            df2["Company"] = c2

            df = pd.concat([df1, df2])

            st.plotly_chart(
                px.line(df, x="Date", y="Close", color="Company"),
                use_container_width=True
            )

        else:
            st.warning("Stock data not available")

        # DRIVERS
        st.subheader("🧠 ESG Drivers")
        col1,col2 = st.columns(2)
        col1.write(generate_drivers(c1,r1))
        col2.write(generate_drivers(c2,r2))

        # INVESTMENT
        st.subheader("📈 Investment Decision")
        col1,col2 = st.columns(2)
        col1.metric(c1, investment_call(s1,0))
        col2.metric(c2, investment_call(s2,0))

    else:
        st.warning("Enter both companies")
